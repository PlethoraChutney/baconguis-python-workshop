def take_order(guest, menu):
    # This line picks a random option from the keys of the
    # given menu. Those will be the names of the dishes.

    order = choice(list(menu.keys()))
    print(guest + ' ordered ' + order)

    # and then we return the price, which is the value for the
    # given key
    return menu[order]

# I'll make my guests using a for loop, but making them by hand
# would work too!

guests = []
for guest in [
    'Alice', 'Bob', 'Charles', 'Danielle',
    'Eddy', 'Frank', 'Greg',
    'Hannah', 'Ingrid', 'James'
    ]:
    guests.append({
        'name': guest,
        'restaurant': choice(list(menus.keys())),
        'order': ''
    })

# not strictly part of the assignment, but will be helpful to make sure everything is working
# can also use it in the last function.
def print_guest(guest):
    # check if order is empty or not
    if guest['order']:
        order = ' ordered ' + guest['order']
    else:
        order = " hasn't ordered yet"
    print(guest['name'] + order + ' from ' + guest['restaurant'] + '.')

print('Before ordering:')
for guest in guests:
    print_guest(guest)

def handle_guests(guest_list, menus):
    total_income = 0.00

    for guest in guest_list:
        menu = menus[guest['restaurant']]
        order = choice(list(menu.keys()))
        guest['order'] = order
        total_income += menu[order]

    return total_income

total_income = handle_guests(guests, menus)

print('\nAfter ordering:')
for guest in guests:
    print_guest(guest)
print('Total sales tonight: ' + str(total_income))
